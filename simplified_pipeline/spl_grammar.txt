SPL_AGENT := "[DEFINE_AGENT:" AGENT_NAME ["\"" STATIC_DESCRIPTION "\""] "]" SPL_PROMPT "[END_AGENT]"

CONCEPTS := [DEFINE_CONCEPTS:] {CONCEPT} [END_CONCEPTS]
CONCEPT := OPTIONAL_ASPECT_NAME ":" STATIC_DESCRIPTION """Concepts do not need parameters or references."""

"""
Data reference design
-) Uniform reference form <REF> ... </REF>
-) SPL compiler analyzes the DESCRIPTION context to determine whether <REF> ... </REF> is use variable value or set variable value
-) Two ways of use variable value:
   - use by reference: <REF> NAME </REF>, VM need to prepend NAME = value before the INSTRUCTION prompt. If NAME refers to file or database, VM needs to load file content or select database and assign the value to a temporary variable NAME.
   - use by value: <REF> *NAME </REF>, i.e., VM retrieve Python variable value and inline variable value at the reference site (used for agent initialization and prompt preparation)
"""

VARIABLES := "[DEFINE_VARIABLES:]" {VARIABLE_DECLARATION} "[END_VARIABLES]"
VARIABLE_DECLARATION := ["\"" DESCRIPTION_WITH_REFERENCES "\""] ["READONLY"] VAR_NAME ":" DATA_TYPE ["=" DEFAULT_VALUE]
VAR_NAME := <word>
DEFAULT_VALUE := VALUE

"""UGAiForge supports the definition of STRUCTURED_DATA_TYPE and ENUM_TYPE as NAMED TYPE"""
TYPES := "[DEFINE_TYPES:]" {ENUM_TYPE_DECLARATION | STRUCTURED_DATA_TYPE_DECLARATION} "[END_TYPES]"
ENUM_TYPE_DECLARATION := ["\"" STATIC_DESCRIPTION "\""] DECLARED_TYPE_NAME "=" ENUM_TYPE
STRUCTURED_DATA_TYPE_DECLARATION := ["\"" STATIC_DESCRIPTION "\""] DECLARED_TYPE_NAME "=" STRUCTURED_DATA_TYPE
DECLARED_TYPE_NAME := <word>

"""UGAiForge stdlib supports commonly applicable operations to each simple data type such as text substring/concatenation, image loading, math operation, time operation, etc."""
DATA_TYPE := ARRAY_DATA_TYPE | STRUCTURED_DATA_TYPE | | ENUM_TYPE | TYPE_NAME | AGENT_NAME
TYPE_NAME := SIMPLE_TYPE_NAME | DECLARED_TYPE_NAME

ENUM_TYPE := "[" <word> {, <word>} "]"
ARRAY_DATA_TYPE := "List [" DATA_TYPE "]"

ELEMENT_NAME := <word>

INSTRUCTION := WORKER_INSTRUCTION
WORKER_INSTRUCTION := "[DEFINE_WORKER:" ["\"" STATIC_DESCRIPTION "\""] WORKER_NAME "]" [INPUTS] [OUTPUTS] MAIN_FLOW {ALTERNATIVE_FLOW} {EXCEPTION_FLOW} [EXAMPLES]"" "[END_WORKER]"
WORKER_NAME := <word>
INSTRUCTION_NAME := WORKER_NAME | GUARDRAIL_NAME

"""
An input or output is required by default if there is no REQUIRED or OPTIONAL annotation.
A worker may use several outputs to achieve the effect of Python Union, for example, one output variable returns the correct response object, and the other returns an error message.
"""
INPUTS := "[INPUTS]" {["REQUIRED" | "OPTIONAL"] [APPLY_GUARDRAILS] [APPLY_CONSTRAINTS] REFERENCE_DATA} "[END_INPUTS]" | "[CONTROLLED_INPUTS]" {["REQUIRED" | "OPTIONAL"] [APPLY_GUARDRAILS] [APPLY_CONSTRAINTS] REFERENCE_DATA} "[END_CONTROLLED_INPUTS]" """REFERENCE should reference DATA_NAME (variable, files, data views)"""
OUTPUTS := "[OUTPUTS]" {["REQUIRED" | "OPTIONAL"] [APPLY_GUARDRAILS] [APPLY_CONSTRAINTS] REFERENCE_DATA} "[END_OUTPUTS]" | "[CONTROLLED_OUTPUT]" {["REQUIRED" | "OPTIONAL"] [APPLY_GUARDRAILS] [APPLY_CONSTRAINTS] REFERENCE_DATA} "[END_CONTROLLED_OUTPUTS]" """REFERENCE should reference DATA_NAME (variable, files, data views)"""

"""
Inputs and outputs of an example should match the INPUTS and OUTPUTS schema.
The example workflow should be a feasible execution path of the declared MAIN_FLOW, ALTERNATIVE_FLOW and/or EXCEPTION_FLOW.
"""

MAIN_FLOW := "[MAIN_FLOW]" {BLOCK} [END_MAIN_FLOW]
ALTERNATIVE_FLOW := "[ALTERNATIVE_FLOW:" CONDITION "]" {BLOCK} "[END_ALTERNATIVE_FLOW]"
EXCEPTION_FLOW := "[EXCEPTION_FLOW:" CONDITION "]" ["LOG" DESCRIPTION_WITH_REFERENCES] {BLOCK} "[END_EXCEPTION_FLOW]"
CONDITION := DESCRIPTION_WITH_REFERENCES

BLOCK := SEQUENTIAL_BLOCK | IF_BLOCK | LOOP_BLOCK
SEQUENTIAL_BLOCK := "[SEQUENTIAL_BLOCK]" {COMMAND} "[END_SEQUENTIAL_BLOCK]"
IF_BLOCK := DECISION_INDEX "[IF" CONDITION "]" {COMMAND} {"[ELSEIF" CONDITION "]" {COMMAND}} ["[ELSE]" {COMMAND}] "END_IF"
LOOP_BLOCK := WHILE_BLOCK | FOR_BLOCK
WHILE_BLOCK := DECISION_INDEX "[WHILE" CONDITION "]" {COMMAND} "[END_WHILE]" """For example, [WHILE not found] do something"""
FOR_BLOCK := DECISION_INDEX "[FOR" CONDITION "]" {COMMAND} [END_FOR] """For example, [For each element in collection] do something"""
DECISION_INDEX := "DECISION-" <number> """System maintained unique decision point index"""

"""
For prompt-to-code commands, UGAiForger generates three code candidates. Users can request more.
UGAiForge allows users to view and select the code candidates (three by default) generated for prompt-to-code commands.
UGAiForge support users to view and test code.
More SYSTEM_COMMAND for text, image, file, math, time operations.
"""

COMMAND := COMMAND_INDEX COMMAND_BODY

COMMAND_BODY := GENERAL_COMMAND |  INVOKE_INSTRUCTION  | REQUEST_INPUT | DISPLAY_MESSAGE
"""If a command is not prompt-to-code or code, the command is a regular prompt."""
GENERAL_COMMAND := "[COMMAND" ["PROMPT_TO_CODE" | "CODE"] DESCRIPTION_WITH_REFERENCES ["STOP" DESCRIPTION_WITH_REFERENCES] ["RESULT" COMMAND_RESULT ["SET" | "APPEND"]] "]" """SET is default operation"""
DISPLAY_MESSAGE := "[DISPLAY" DESCRIPTION_WITH_REFERENCES "]"
REQUEST_INPUT := "[INPUT" ["DISPLAY"] DESCRIPTION_WITH_REFERENCE "VALUE" COMMAND_RESULT ["SET" | "APPEND"] [APPLY_GUARDRAILS] "]" """With DISPLAY, DESCRIPTION_WITH_REFERENCE is regarded as display text. Otherwise, DESCRIPTION_WITH_REFERENCE is a prompt."""
INVOKE_INSTRUCTION := "[INVOKE" INSTRUCTION_NAME {"," INSTRUCTION_NAME} ["WITH" ARGUMENT_LIST {"," ARGUMENT_LIST}] ["RESPONSE" COMMAND_RESULT ["SET" | "APPEND"]]  "]"
INSTRUCTION_NAME := WORKER_NAME 
ARGUMENT_LIST := STRUCTURED_TEXT

COMMAND_RESULT := VAR_NAME ":" DATA_TYPE | REFERENCE """VAR_NAME ":" DATA_TYPE essentially declares a temporary variable"""


EXAMPLES := "[EXAMPLES]" { EXPECTED_WORKER_BEHAVIOR | DEFECT_WORKER_BEHAVIOR } "[END_EXAMPLES]"

"""
An EXPECTED_WORKER_BEHAVIOR is meant to illustrate how the worker should correctly respond (i.e., expected behavior and output ) to both valid and invalid inputs (if applicable) as per its defined logic.
"""
EXPECTED_WORKER_BEHAVIOR := "<EXPECTED-WORKER-BEHAVIOR>" "{" EXPECTED_WORKER_BEHAVIOR_DETAILS "}" "</EXPECTED-WORKER-BEHAVIOR>"
EXPECTED_WORKER_BEHAVIOR_DETAILS :=  INPUT_EXAMPLE"," EXPECTED_OUTPUT_EXAMPLE "," EXECUTION_PATH
INPUT_EXAMPLE := "inputs:" "{" VAR_VALUE_PAIRS "}" """The input parameters provided to the worker"""
VAR_VALUE_PAIRS := VAR_NAME ":" VALUE {"," VAR_VALUE_PAIRS}
EXPECTED_OUTPUT_EXAMPLE := "expected-outputs:" "{" VAR_VALUE_PAIRS "}" """Specifies the expected correct outputs for the given inputs"""
EXECUTION_PATH := "execution-path:" COMMAND_DECISION_INDEX {"," COMMAND_DECISION_INDEX} """Documents the sequence of commands and decisions (e.g., C-1, D-1, C-3) followed within the worker's logic to produce the outputs"""
COMMAND_DECISION_INDEX := COMMAND_INDEX | DECISION_INDEX

"""
A DEFECT_WORKER_BEHAVIOR is meant to illustrate how the worker malfunctions and thus incorrectly respond (i.e., erroneous behavior and output ) to both valid and invalid inputs (if applicable) as per its defined logic.

DEFECT_TYPE definition:
-) Processing Logic Error pertains to the core operational logic, where the system fails to correctly process inputs due to logic flaws.
-) Error Handling Error relates to the system's inability to correctly identify and respond to invalid inputs or states.
-) Output Generation Error involves issues with producing the correct output format or content, even if the input was correctly processed.
-) Input Specification Error involves failures in meeting the worker's input specifications
"""
DEFECT_WORKER_BEHAVIOR := "<DEFECT-WORKER-BEHAVIOR>" "{" DEFECT_WORKER_BEHAVIOR_DETAILS "}" "</DEFECT-WORKER-BEHAVIOR>""
DEFECT_WORKER_BEHAVIOR_DETAILS := DEFECT_TYPE "," INPUT_EXAMPLE"," DEFECT_OUTPUT_EXAMPLE "," EXECUTION_PATH "," DEFECT_EXPLANATION
MALFUNCTION_TYPE := "Processing Logic Error" | "Invalid Input Handling Error" | "Output Generation Error" | "Input Specification Error" """"Types of defects in the worker's operation that result in the errors"""
DEFECT_OUTPUT_EXAMPLE := "defect-outputs:" "{" VAR_VALUE_PAIRS "}" """The outputs that incorrectly result from the defect"""
DEFECT_EXPLANATION := "defect-explanation:" STATIC_DESCRIPTION """A short description of why the output is incorrect given the specific defect."""


DESCRIPTION_WITH_REFERENCES := STATIC_DESCRIPTION {DESCRIPTION_WITH_REFERENCES} | REFERENCE {DESCRIPTION_WITH_REFERENCES}
STATIC_DESCRIPTION := <word> | <word> <space> STATIC_DESCRIPTION

NAME := SIMPLE_NAME | QUALIFIED_NAME | ARRAY_ACCESS | DICT_ACCESS
